## ğŸ¯ Key Algorithms

### Room Assignment Algorithm
```javascript
function findOrCreateRoom(username) {
  // Algorithm: First-fit room allocation
  
  for (let [roomId, room] of chatRooms) {
    if (room.users.length < 2) {  // Found available room
      room.users.push({socketId, username});
      return roomId;
    }
  }
  
  // No available room, create new
  const newRoomId = generateUniqueId();
  chatRooms.set(newRoomId, {
    id: newRoomId,
    users: [{socketId, username}],
    messages: []
  });
  return newRoomId;
}
```

### Message Broadcasting Algorithm
```javascript
function broadcastMessage(roomId, message) {
  // Algorithm: Room-scoped broadcast
  
  const room = chatRooms.get(roomId);
  
  // Send only to users in this specific room
  io.to(roomId).emit('receive_message', {
    username: message.username,
    text: message.text,
    timestamp: message.timestamp
  });
  
  // Store for history
  room.messages.push(message);
}
```
